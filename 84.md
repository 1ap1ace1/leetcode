```c++
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int mmax = 0;
        for(int i = 0;i < heights.size(); ++i){
            int sum = 0;
            if(heights[i] == 0)
                continue;
            for(int j = 0;i - j >=0; ++j){
                if(heights[i - j] < heights[i])
                    break;
                else
                    sum += heights[i];
            }
            // cout << sum <<';';
            for(int j = 1;i + j < heights.size(); ++j){
                if(heights[i + j] < heights[i])
                    break;
                else
                    sum += heights[i];
            }
            // cout << sum <<endl;
            if (sum > mmax)
                mmax = sum;
        }
        return mmax;
    }
};
```
