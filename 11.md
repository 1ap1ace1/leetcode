class Solution:
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        mmax = 0
        i = 0
        j = len(height) - 1
        while(i < j):
            if ( min(height[i],height[j])*(j - i) > mmax ):
                mmax = min(height[i],height[j])*(j - i)
            if (height[i] < height[j]):
                for k in range(1 , len(height)):
                    if i + k >= j:
                        i = i + k
                        break
                    if height[i] < height[i + k]:
                        i = i + k
                        break
            else:
                for k in range(1 , len(height)):
                    if i >= j - k:
                        j = j - k
                        break
                    if height[j] < height[j - k]:
                        j = j - k
                        break
        return mmax
