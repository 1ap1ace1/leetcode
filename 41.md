class Solution(object):
    def firstMissingPositive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        dic = dict()
        for i in range(len(nums)):
            if nums[i] > 0:
                dic[nums[i]] = 1
        j = 1
        while j > 0:
            if j not in dic.keys():
                return j
            j += 1
